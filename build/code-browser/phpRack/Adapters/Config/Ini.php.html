  95.  95                 $currentElement =& $currentElement[$keyFragment];
  96.  96             }

  98.  98         }

  100. 100     }

  102. 102     /**

  113. 113             throw new Exception("INI file '{$filename}' doesn't exist");
  114. 114         }
  115. 115         $sections = array();

  117. 117 
  118. 118         foreach ($iniFileSections as $sectionName => $data) {
  119. 119             // divide section name to check it have some parent ([section : parent])
  120. 120             $nameParts = explode(':', $sectionName);
  121. 121             $thisSectionName = trim($nameParts[0]);
  122. 122 
  123. 123             // if section have parent
  124. 124             if (isset($nameParts[1])) {
  125. 125                 $parentSectionName = trim($nameParts[1]);
  126. 126                 // merge current section values, with parent values
  127. 127                 $data = array_merge(
  128. 128                     $sections[$parentSectionName],
  129. 129                     $data
  130. 130                 );
  131. 131             }
  132. 132             $sections[$thisSectionName] = $data;
  133. 133         }
  134. 134 
  135. 135         return $sections;
  136. 136     }
